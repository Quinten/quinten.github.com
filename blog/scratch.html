# Start a simple web server in the current directory
while true; do { echo -e 'HTTP/1.1 200 OK\r\n'; cat index.html; } | nc -l 8000; done
# Open the web server in lynx
lynx http://127.0.0.1:8000
# The html is not parsed, you see the source code
# But nc outputs the request, so you can see the request headers
GET / HTTP/1.0
Host: 127.0.0.1:8000
Accept: text/html, text/plain, text/sgml, text/css, application/xhtml+xml, */*;q=0.01
Accept-Encoding: gzip, compress, bzip2
Accept-Language: en
User-Agent: Lynx/2.8.9rel.1 libwww-FM/2.14 SSL-MM/1.4.1 OpenSSL/3.2.1
